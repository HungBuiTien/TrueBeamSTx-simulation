Step 1: Install cluster tools

	sudo apt-get install htcondor -y
	sudo condor_master

Step 2: Add environment variables
	Open file .bashrc
	add 2 lines:
		+ export GC_DOT_GATE_DIR=/home/gate/Softwares/Gate
		+ export GC_GATE_EXE_DIR=/home/gate/Softwares/Gate/Gate-install/bin

Step 3: Build and Copy files to ../bin
	cd /home/gate/Softwares/Gate/Gate/cluster_tools/jobsplitter
	make
	sudo cp -rf gjs /home/gate/Softwares/Gate/Gate-install/bin
	cd /home/gate/Softwares/Gate/Gate/cluster_tools/filemerger
	make
	sudo cp -rf gjm /home/gate/Softwares/Gate/Gate-install/bin

Step 4: Turn on the DAQClusterStart

Step 5: Create a submition files
	/home/gate/Softwares/Gate/Gate/cluster_tools/jobsplitter/gjs -numberofsplits 10 -clusterplatform condor -condorscript /home/gate/Softwares/Gate/Gate/cluster_tools/jobsplitter/script/condor.script mac/S1Main.mac
	/home/gate/Softwares/Gate/Gate/cluster_tools/jobsplitter/gjs -numberofsplits 500 -clusterplatform condor -condorscript /home/gate/Softwares/Gate/Gate/cluster_tools/jobsplitter/script/condor.script mac/S2Main.mac

Step 6: Submit files
	cd your_dir
	condor_submit mac/S1Main.submit
	condor_submit mac/S2Main.submit

Step 7: Check status
	condor_q

Step 8: Sum-up results
